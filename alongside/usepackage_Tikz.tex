% ---- arrow macros
\newcommand{\arcarrow}[4]{%
   % #1 = start angle
   % #2 = end angle
   % #3 = text
   % #4 = fill color
   \pgfmathsetmacro{\rin}{1.7}
   \pgfmathsetmacro{\rmid}{2.2}
   \pgfmathsetmacro{\rout}{2.7}
   \pgfmathsetmacro{\astart}{#1}
   \pgfmathsetmacro{\aend}{#2}
   \pgfmathsetmacro{\atip}{5} % tip protrusion angle

   % Draw the arrow
   \fill[#4, very thick] (\astart+\atip:\rin)
       arc (\astart+\atip:\aend:\rin)
         -- (\aend-\atip:\rmid)
         -- (\aend:\rout)
       arc (\aend:\astart+\atip:\rout)
         -- (\astart:\rmid)
       -- cycle;

   % Place text along the middle arc
   \path[
      decoration = {
         text along path,
         text = {|\mytextstyle|#3},
         text align = center,
         raise = -1.0ex
      },
      decorate
   ](\astart+\atip:\rmid)
     arc (\astart+\atip:\aend+\atip:\rmid);
}

\newcommand{\arcarrowouter}[4]{%
   % #1 = start angle
   % #2 = end angle
   % #3 = text
   % #4 = fill color
   \pgfmathsetmacro{\rin}{2.8}
   \pgfmathsetmacro{\rmid}{3.3}
   \pgfmathsetmacro{\rout}{3.8}
   \pgfmathsetmacro{\astart}{#1}
   \pgfmathsetmacro{\aend}{#2}
   \pgfmathsetmacro{\atip}{5}

   \fill[#4, very thick] (\astart+\atip:\rin)
       arc (\astart+\atip:\aend:\rin)
         -- (\aend-\atip:\rmid)
         -- (\aend:\rout)
       arc (\aend:\astart+\atip:\rout)
         -- (\astart:\rmid)
       -- cycle;

   \path[
      decoration = {
         text along path,
         text = {|\mytextstyle|#3},
         text align = center,
         raise = -1.0ex
      },
      decorate
   ](\astart+\atip:\rmid)
     arc (\astart+\atip:\aend+\atip:\rmid);
}
